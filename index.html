🗄 Configuração do Banco de Dados

Supabase.com → seu projeto → Settings → API

SQL para criar a tabela (execute uma vez no Supabase)
CREATE TABLE IF NOT EXISTS mro_oportunidades ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, created_at timestamptz DEFAULT now(), cliente text NOT NULL, mes text, ano text, numero text NOT NULL, portal text, valor numeric, status text, participamos text, venda text, data_pedido date, obs text ); ALTER TABLE mro_oportunidades ENABLE ROW LEVEL SECURITY; -- Política mais segura: somente usuários autenticados CREATE POLICY "Allow authenticated" ON mro_oportunidades FOR ALL TO authenticated USING (true) WITH CHECK (true); -- Se precisar de acesso anon (sem login), use esta em vez da de cima: -- CREATE POLICY "Allow all" ON mro_oportunidades -- FOR ALL TO anon, authenticated -- USING (true) WITH CHECK (true);
